    export function getExpandedIds(
      prev: Set<string>,
      taskId: string,
      shiftKey: boolean
    ): Set<string> {
      const next = new Set(prev);
      if (shiftKey) {
        next.has(taskId) ? next.delete(taskId) : next.add(taskId);
      } else {
        if (next.has(taskId)) { next.clear(); }
        else { next.clear(); next.add(taskId); }
      }
