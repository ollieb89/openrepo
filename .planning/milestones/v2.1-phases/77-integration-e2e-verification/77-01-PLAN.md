```python
# Helper
# ---------------------------------------------------------------------------

async def _collect_events(sock_path: str, n: int, timeout: float = 3.0) -> List[dict]:
    """Connect to the bridge socket and collect n newline-delimited JSON events."""
    collected = []
    deadline = time.monotonic() + timeout

    while time.monotonic() < deadline:
        try:
            reader, writer = await asyncio.open_unix_connection(sock_path)
            break
        except (FileNotFoundError, ConnectionRefusedError):
            await asyncio.sleep(0.05)
    else:
        raise RuntimeError(f"Could not connect to bridge socket at {sock_path} within {timeout}s")

    try:
        while len(collected) < n and time.monotonic() < deadline:
            remaining = deadline - time.monotonic()
            try:
                line = await asyncio.wait_for(reader.readline(), timeout=remaining)
            except asyncio.TimeoutError:
                break
            if not line:
                break
            line = line.decode("utf-8", errors="replace").strip()
            if line:
                try:
                    collected.append(json.loads(line))
                except json.JSONDecodeError:
                    pass  # skip malformed lines
    finally:
        writer.close()
        try:
            await writer.wait_closed()
        except Exception:
            pass

    return collected

@pytest.mark.asyncio
async def test_task_lifecycle_events_flow_in_order(sock_path):
    """Full task lifecycle: created → started → output → completed flows through bridge.

    Validates INTG-01: event stream shows no gaps — all expected event types
    (task.created, task.started, task.output, task.completed) appear in correct order.
    """
    from openclaw.events.bridge import ensure_event_bridge

    ensure_event_bridge()

    # Brief pause for bridge to start listening
    await asyncio.sleep(0.1)

    # Start collector (runs concurrently, collects 4 events)
    collect_task = asyncio.create_task(_collect_events(sock_path, n=4, timeout=3.0))

    # Give the collector a moment to connect
    await asyncio.sleep(0.05)

    # Emit full task lifecycle for project "test-project", task "task-e2e-001"
    project_id = "test-project"
    task_id = "task-e2e-001"

    emit({
        "event_type": EventType.TASK_CREATED.value,
        "domain": EventDomain.TASK.value,
        "project_id": project_id,
        "task_id": task_id,
        "payload": {"skill_hint": "code"},
        "timestamp": time.time(),
    })
    await asyncio.sleep(0.02)

    emit({
        "event_type": EventType.TASK_STARTED.value,
        "domain": EventDomain.TASK.value,
        "project_id": project_id,
        "task_id": task_id,
        "payload": {"container_name": f"openclaw-{project_id}-l3-{task_id}"},
        "timestamp": time.time(),
    })
    await asyncio.sleep(0.02)

    emit({
        "event_type": EventType.TASK_OUTPUT.value,
        "domain": EventDomain.TASK.value,
        "project_id": project_id,
        "task_id": task_id,
        "payload": {"line": "Hello from L3 container", "stream": "stdout"},
        "timestamp": time.time(),
    })
    await asyncio.sleep(0.02)

    emit({
        "event_type": EventType.TASK_COMPLETED.value,
        "domain": EventDomain.TASK.value,
        "project_id": project_id,
        "task_id": task_id,
        "payload": {"exit_code": 0},
        "timestamp": time.time(),
    })

    events = await collect_task

    # Verify all 4 event types arrived
    assert len(events) == 4, (
        f"Expected 4 events in pipeline, got {len(events)}: {[e.get('event_type') for e in events]}"
    )

    # Verify correct ordering
    event_types = [e.get("event_type") for e in events]
    assert event_types == [
        EventType.TASK_CREATED.value,
        EventType.TASK_STARTED.value,
        EventType.TASK_OUTPUT.value,
        EventType.TASK_COMPLETED.value,
    ], f"Events arrived out of order: {event_types}"

    # Verify task_id is preserved in each event
    for evt in events:
        assert evt.get("task_id") == task_id, (
            f"task_id missing or wrong in event: {evt}"
        )
```

```python
"""
Integration test: metrics reflect full task lifecycle state (INTG-01).

Verifies that after a task completes, collect_metrics_from_state() returns
updated completed_count. No Docker or gateway required.

Run:
    uv run pytest packages/orchestration/tests/test_metrics_lifecycle.py -v
"""
import json
import time
from pathlib import Path
from unittest.mock import patch

import pytest

from openclaw.state_engine import JarvisState


# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------

def _make_state_file(tmp_path: Path, tasks: dict) -> Path:
    """Write a workspace-state.json with the given tasks dict and return the path."""
    state = {"tasks": tasks, "version": 1}
    p = tmp_path / "workspace-state.json"
    p.write_text(json.dumps(state))
    return p


# ---------------------------------------------------------------------------
# Tests
# ---------------------------------------------------------------------------

class TestMetricsLifecycle:
    """Verify metrics snapshot reflects live task state."""

    def test_completed_task_increments_metrics_count(self, tmp_path):
        """After a task completes, metrics show completed_count >= 1."""
        from openclaw.metrics import collect_metrics_from_state

        state_file = _make_state_file(tmp_path, {})

        with patch("openclaw.config.get_project_root", return_value=tmp_path):
            with patch("openclaw.config.get_state_path", return_value=state_file):
                # Create and complete a task via JarvisState
                jarvis = JarvisState(state_file)
                jarvis.create_task("task-complete-001", "code", metadata={})
                jarvis.update_task("task-complete-001", "completed")

                # Load state dict directly (mirrors how metrics endpoint works)
                state = json.loads(state_file.read_text())
                metrics = collect_metrics_from_state(state)

        completed = metrics.get("completed_tasks", metrics.get("tasks_completed", 0))
        assert completed >= 1, (
            f"Expected at least 1 completed task in metrics, got: {metrics}"
        )

    def test_in_progress_task_shows_in_active_count(self, tmp_path):
        """An in-progress task appears in active_tasks metric."""
        from openclaw.metrics import collect_metrics_from_state

        state_file = _make_state_file(tmp_path, {})

        with patch("openclaw.config.get_project_root", return_value=tmp_path):
            with patch("openclaw.config.get_state_path", return_value=state_file):
                jarvis = JarvisState(state_file)
                jarvis.create_task("task-active-001", "code", metadata={})
                jarvis.update_task("task-active-001", "in_progress")

                state = json.loads(state_file.read_text())
                metrics = collect_metrics_from_state(state)

        active = metrics.get("active_tasks", metrics.get("tasks_in_progress", 0))
        assert active >= 1, (
            f"Expected at least 1 active task in metrics, got: {metrics}"
        )

    def test_full_lifecycle_metrics_progression(self, tmp_path):
        """Full lifecycle: pending → in_progress → completed shows correct counts at each stage."""
        from openclaw.metrics import collect_metrics_from_state

        state_file = _make_state_file(tmp_path, {})

        with patch("openclaw.config.get_project_root", return_value=tmp_path):
            with patch("openclaw.config.get_state_path", return_value=state_file):
                jarvis = JarvisState(state_file)
                jarvis.create_task("lifecycle-task", "code", metadata={})

                # Stage 1: pending
                state = json.loads(state_file.read_text())
                m1 = collect_metrics_from_state(state)

                # Stage 2: in_progress
                jarvis.update_task("lifecycle-task", "in_progress")
                state = json.loads(state_file.read_text())
                m2 = collect_metrics_from_state(state)

                # Stage 3: completed
                jarvis.update_task("lifecycle-task", "completed")
                state = json.loads(state_file.read_text())
                m3 = collect_metrics_from_state(state)

        # At completion, total_tasks should be >= 1
        total = m3.get("total_tasks", m3.get("tasks_total", 0))
        completed = m3.get("completed_tasks", m3.get("tasks_completed", 0))
        assert total >= 1, f"total_tasks should be >= 1 after lifecycle: {m3}"
        assert completed >= 1, f"completed_tasks should be >= 1 after completion: {m3}"
```
