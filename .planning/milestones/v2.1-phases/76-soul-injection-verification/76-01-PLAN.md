# Phase 76: SOUL Injection Verification Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Verify and fix that every L3 container spawned has its SOUL variables fully populated — including `active_task_count`, `pool_utilization`, and topology context (archetype, agent count, confidence) — and add tests proving these values are non-empty and reflect live state.

**Architecture:** The SOUL injection pipeline runs in `spawn.py:_build_augmented_soul()` → `soul_renderer.build_dynamic_variables()` → template substitution in `soul_renderer.render_soul()`. The gap: `soul-default.md` template lacks `$topology_archetype`, `$topology_agent_count`, `$topology_confidence` placeholders, so computed values are silently dropped. This plan adds them to the template and writes integration tests that verify populated SOUL files using a real `JarvisState` + `TopologyGraph` (no Docker required).

**Tech Stack:** Python 3.11+, pytest, uv (test runner), `openclaw` package at `packages/orchestration/src/`

```python
class TestSOULPopulationIntegration:
    """Integration tests: SOUL populated from real JarvisState + topology on disk.

    Uses real JarvisState and topology storage (no mocking of data layer)
    but patches get_project_root() to redirect all file I/O to tmp_path.
    Validates OBSV-03 success criteria without requiring Docker.
    """

    @pytest.fixture
    def project_env(self, tmp_path):
        """Minimal project structure: project config, state dir, agent configs."""
        # Template — copy the real soul-default.md so topology section is present
        real_template = (
            Path(__file__).parent.parent.parent.parent
            / "agents" / "_templates" / "soul-default.md"
        )
        template_dir = tmp_path / "agents" / "_templates"
        template_dir.mkdir(parents=True)
        (template_dir / "soul-default.md").write_text(real_template.read_text())

        # Project config
        project_dir = tmp_path / "projects" / "testproj"
        project_dir.mkdir(parents=True)
        project_config = {
            "id": "testproj",
            "name": "Test Project",
            "agents": {"l2_pm": "test_pm"},
            "tech_stack": {"frontend": "React", "backend": "Python", "infra": "Docker"},
        }
        (project_dir / "project.json").write_text(json.dumps(project_config))

        # State directory
        state_dir = tmp_path / "workspace" / ".openclaw" / "testproj"
        state_dir.mkdir(parents=True)
        (state_dir / "workspace-state.json").write_text(json.dumps({"tasks": {}}))

        # Agent configs for AgentRegistry auto-discovery
        pm_dir = tmp_path / "agents" / "test_pm" / "agent"
        pm_dir.mkdir(parents=True)
        (pm_dir / "config.json").write_text(json.dumps({
            "id": "test_pm", "name": "Test PM", "level": 2,
            "reports_to": "clawdia_prime", "max_concurrent": 3, "skills": [],
        }))

        l3_dir = tmp_path / "agents" / "l3_specialist" / "agent"
        l3_dir.mkdir(parents=True)
        (l3_dir / "config.json").write_text(json.dumps({
            "id": "l3_specialist", "name": "L3 Specialist", "level": 3,
            "reports_to": "test_pm", "max_concurrent": 1, "skills": [],
        }))

        return tmp_path

    def test_active_task_count_nonzero_when_task_in_progress(self, project_env):
        """active_task_count reflects in-progress tasks from real JarvisState."""
        from openclaw.state_engine import JarvisState
        from openclaw.agent_registry import AgentRegistry

        state_file = project_env / "workspace" / ".openclaw" / "testproj" / "workspace-state.json"

        with patch("openclaw.config.get_project_root", return_value=project_env):
            with patch("openclaw.soul_renderer._find_project_root", return_value=project_env):
                with patch("openclaw.config.get_state_path", return_value=state_file):
                    jarvis = JarvisState(state_file)
                    jarvis.create_task("task-001", "code", metadata={})
                    jarvis.update_task("task-001", "in_progress")

                    registry = AgentRegistry(project_env)
                    vars = build_dynamic_variables("testproj", "l3_specialist", registry)

                    assert vars["active_task_count"] == "1", (
                        f"active_task_count should be 1, got {vars['active_task_count']!r}"
                    )
                    assert vars["pool_utilization"] == "1/1", (
                        f"pool_utilization should be 1/1, got {vars['pool_utilization']!r}"
                    )

    def test_two_concurrent_states_show_different_counts(self, project_env):
        """Different JarvisState contents produce different active_task_count values.

        Validates OBSV-03 criterion 2: two concurrent L3 tasks show different counts.
        We test this by calling build_dynamic_variables twice with different state on disk.
        """
        from openclaw.state_engine import JarvisState
        from openclaw.agent_registry import AgentRegistry

        state_file = project_env / "workspace" / ".openclaw" / "testproj" / "workspace-state.json"

        with patch("openclaw.config.get_project_root", return_value=project_env):
            with patch("openclaw.soul_renderer._find_project_root", return_value=project_env):
                with patch("openclaw.config.get_state_path", return_value=state_file):
                    registry = AgentRegistry(project_env)

                    # Before any tasks: count is 0
                    vars_before = build_dynamic_variables("testproj", "l3_specialist", registry)
                    count_before = vars_before["active_task_count"]

                    # Add an in-progress task
                    jarvis = JarvisState(state_file)
                    jarvis.create_task("task-001", "code", metadata={})
                    jarvis.update_task("task-001", "in_progress")

                    # After adding: count is 1
                    vars_after = build_dynamic_variables("testproj", "l3_specialist", registry)
                    count_after = vars_after["active_task_count"]

                    assert count_before != count_after, (
                        "Expected different active_task_count for 0 vs 1 in-progress tasks"
                    )
                    assert count_after == "1"

    def test_topology_context_in_rendered_soul_after_save(self, project_env):
        """After saving a topology, rendered SOUL contains archetype and agent count.

        Validates OBSV-03 criterion 3: after proposing a topology, spawned L3
        containers have topology archetype and agent count in SOUL context.
        """
        from openclaw.state_engine import JarvisState
        from openclaw.agent_registry import AgentRegistry
        from openclaw.topology.models import TopologyGraph, TopologyNode
        from openclaw.topology.storage import save_topology
        from openclaw.soul_renderer import render_soul

        state_file = project_env / "workspace" / ".openclaw" / "testproj" / "workspace-state.json"

        with patch("openclaw.config.get_project_root", return_value=project_env):
            with patch("openclaw.soul_renderer._find_project_root", return_value=project_env):
                with patch("openclaw.topology.storage.get_project_root", return_value=project_env):
                    with patch("openclaw.config.get_state_path", return_value=state_file):
                        # Save a topology with a known archetype
                        graph = TopologyGraph(
                            project_id="testproj",
                            nodes=[
                                TopologyNode(id="a1", level=1, intent="coordinate", risk_level="low"),
                                TopologyNode(id="a2", level=3, intent="execute", risk_level="medium"),
                                TopologyNode(id="a3", level=3, intent="execute", risk_level="medium"),
                            ],
                            edges=[],
                            metadata={"archetype": "balanced", "structural_confidence": 0.82},
                        )
                        save_topology("testproj", graph)

                        registry = AgentRegistry(project_env)
                        dynamic_vars = build_dynamic_variables("testproj", "l3_specialist", registry)

                        # Verify dynamic variable values
                        assert dynamic_vars["topology_archetype"] == "balanced", (
                            f"Expected archetype=balanced, got {dynamic_vars['topology_archetype']!r}"
                        )
                        assert dynamic_vars["topology_agent_count"] == "3", (
                            f"Expected 3 nodes, got {dynamic_vars['topology_agent_count']!r}"
                        )
                        assert dynamic_vars["topology_confidence"] == "0.82", (
                            f"Expected confidence=0.82, got {dynamic_vars['topology_confidence']!r}"
                        )

                        # Verify values appear in the rendered SOUL output
                        from openclaw.soul_renderer import build_variables
                        from openclaw.project_config import load_project_config
                        with patch("openclaw.soul_renderer.load_project_config") as mock_cfg:
                            mock_cfg.return_value = {
                                "id": "testproj", "name": "Test Project",
                                "agents": {"l2_pm": "test_pm"},
                                "tech_stack": {"frontend": "React", "backend": "Python", "infra": "Docker"},
                            }
                            all_vars = {**dynamic_vars, "memory_section": "No context loaded."}
                            soul = render_soul("testproj", extra_variables=all_vars)

                        assert "balanced" in soul, "topology_archetype not rendered in SOUL"
                        assert "Structural Confidence" in soul, "Topology Context section missing from SOUL"

    def test_soul_template_has_topology_placeholders(self):
        """Regression guard: soul-default.md must contain topology placeholder variables.

        If someone removes the topology section from the template, the dynamic
        variables will be silently dropped from SOUL. This test prevents that regression.
        """
        template_path = (
            Path(__file__).parent.parent.parent.parent
            / "agents" / "_templates" / "soul-default.md"
        )
        content = template_path.read_text()
        assert "$topology_archetype" in content, "soul-default.md missing $topology_archetype"
        assert "$topology_agent_count" in content, "soul-default.md missing $topology_agent_count"
        assert "$topology_confidence" in content, "soul-default.md missing $topology_confidence"
```
