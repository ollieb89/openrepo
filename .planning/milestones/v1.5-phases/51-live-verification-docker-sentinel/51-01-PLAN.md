Write `/tmp/51-test-sigterm-drain.py` with the following content:

```python
"""
Test 2: SIGTERM drain verification for pool.py.

Strategy (Option A from research): manually inject a slow asyncio coroutine into
_pending_memorize_tasks to create a deterministic timing window, then SIGTERM
the process itself after 0.5s while the task is in-flight.

Uses a unique timestamp marker in the memorize payload to avoid false positives
from previous test runs.
"""
import asyncio
import signal
import os
import sys
import time
import json

sys.path.insert(0, '~/.openclaw/packages/orchestration/src')
sys.path.insert(0, '~/.openclaw/skills/spawn')
os.environ['OPENCLAW_ROOT'] = '~/.openclaw'

from pool import L3ContainerPool, register_shutdown_handler

# Unique marker for this test run
TEST_MARKER = f"test-sigterm-drain-marker-{int(time.time())}"
print(f"Test marker: {TEST_MARKER}", flush=True)

# Write marker to a file so the verification step can retrieve it
with open('/tmp/51-test2-marker.txt', 'w') as f:
    f.write(TEST_MARKER)


async def slow_memorize_task():
    """
    Simulates a memorize call that takes 3 seconds before completing.
    After the artificial delay, calls memU /memorize to store the test marker.
    """
    print("Memorize task started (3s sleep to simulate in-flight call)...", flush=True)
    await asyncio.sleep(3)  # Creates reliable timing window for SIGTERM

    # Call memU to store the marker — confirms drain waited for this
    import urllib.request
    payload = json.dumps({
        'resource_url': TEST_MARKER,
        'resource_description': 'SIGTERM drain verification — this entry confirms drain completed',
        'user': {'user_id': 'pumplai'},
    }).encode('utf-8')
    req = urllib.request.Request(
        'http://localhost:18791/memorize',
        data=payload,
        headers={'Content-Type': 'application/json'},
        method='POST',
    )
    try:
        with urllib.request.urlopen(req, timeout=10) as resp:
            print(f"memU memorize response: {resp.status}", flush=True)
    except Exception as e:
        print(f"ERROR: memU memorize failed: {e}", flush=True)
        raise

    print("Memorize task completed successfully", flush=True)


async def main():
    pool = L3ContainerPool(max_concurrent=3, project_id='pumplai')

    # Inject the slow memorize task (simulates post-container fire-and-forget)
    task = asyncio.create_task(slow_memorize_task())
    pool._pending_memorize_tasks.append(task)
    print("Injected slow memorize task into _pending_memorize_tasks", flush=True)

    # Register SIGTERM handler (same path as production)
    loop = asyncio.get_running_loop()
    register_shutdown_handler(loop, pool)
    print("SIGTERM handler registered", flush=True)

    # Send SIGTERM to ourselves after 0.5s (memorize task is in-flight at 3s, SIGTERM at 0.5s)
    async def send_sigterm():
        await asyncio.sleep(0.5)
        print("Sending SIGTERM to self...", flush=True)
        os.kill(os.getpid(), signal.SIGTERM)

    asyncio.create_task(send_sigterm())

    # Run until drain handler calls loop.stop()
    # The drain will await the 3s memorize task before stopping
    try:
        await asyncio.sleep(60)  # Will be interrupted by drain handler
    except Exception:
        pass  # Expected — loop.stop() raises RuntimeError or CancelledError

    print("main() returning", flush=True)


if __name__ == '__main__':
    asyncio.run(main())
    print("Process exiting cleanly after drain", flush=True)
```
