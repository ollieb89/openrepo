**Step 3 — Inject SSE listener and timing measurement via browser_evaluate:**

Inject JavaScript that:
- Opens an EventSource connection to `/occc/api/events?project=pumplai`
- Records `window.__sse_t0` when connection is opened
- Records `window.__sse_task_created_time` when `task.created` event arrives
- Sets `window.__sse_task_id` to the received task ID
- Stores all received events in `window.__sse_events`

```playwright
mcp__playwright__browser_evaluate: {
  script: "
    window.__sse_events = [];
    window.__sse_task_created_time = null;
    window.__sse_task_id = null;
    window.__sse_t0 = Date.now();
    const es = new EventSource('/occc/api/events?project=pumplai');
    window.__sse_source = es;
    es.addEventListener('message', (e) => {
      const now = Date.now();
      try {
        const data = JSON.parse(e.data);
        window.__sse_events.push({ type: data.type, time: now, elapsed: now - window.__sse_t0 });
        if (data.type === 'task.created' && !window.__sse_task_created_time) {
          window.__sse_task_created_time = now;
          window.__sse_task_id = data.task_id;
        }
      } catch(err) {}
    });
    es.onerror = (e) => { window.__sse_error = true; };
    'SSE listener installed at T0=' + window.__sse_t0;
  "
}
```
